  9 /// mock_dependencies is a drop-in replacement for cosmwasm_std::testing::mock_dependencies
 10 /// this uses our CustomQuerier.
 11 pub fn mock_dependencies(
 12     canonical_length: usize,
 13     contract_balance: &[Coin],
 14 ) -> Extern<MockStorage, MockApi, WasmMockQuerier> {
 15     let contract_addr = HumanAddr::from(MOCK_CONTRACT_ADDR);
 16     let custom_querier: WasmMockQuerier =
 17         WasmMockQuerier::new(MockQuerier::new(&[(&contract_addr, contract_balance)]));
 18 
 19     Extern {
 20         storage: MockStorage::default(),
 21         api: MockApi::new(canonical_length),
 22         querier: custom_querier,
 23     }
 24 }

 31 #[derive(Clone, Default)]
 32 pub struct TokenQuerier {
 33     // this lets us iterate over all pairs that match the first string
 34     balances: HashMap<HumanAddr, HashMap<HumanAddr, Uint128>>,
 35 }
 36 
 37 impl TokenQuerier {
 38     pub fn new(balances: &[(&HumanAddr, &[(&HumanAddr, &Uint128)])]) -> Self {
 39         TokenQuerier {
 40             balances: balances_to_map(balances),
 41         }
 42     }
 43 }
 44 
 45 pub(crate) fn balances_to_map(
 46     balances: &[(&HumanAddr, &[(&HumanAddr, &Uint128)])],
 47 ) -> HashMap<HumanAddr, HashMap<HumanAddr, Uint128>> {
 48     let mut balances_map: HashMap<HumanAddr, HashMap<HumanAddr, Uint128>> = HashMap::new();
 49     for (contract_addr, balances) in balances.iter() {
 50         let mut contract_balances_map: HashMap<HumanAddr, Uint128> = HashMap::new();
 51         for (addr, balance) in balances.iter() {
 52             contract_balances_map.insert(HumanAddr::from(addr), **balance);
 53         }
 54 
 55         balances_map.insert(HumanAddr::from(contract_addr), contract_balances_map);
 56     }
 57     balances_map
 58 }

 60 impl Querier for WasmMockQuerier {
 61     fn raw_query(&self, bin_request: &[u8]) -> QuerierResult {
 62         // MockQuerier doesn't support Custom, so we ignore it completely here
 63         let request: QueryRequest<Empty> = match from_slice(bin_request) {
 64             Ok(v) => v,
 65             Err(e) => {
 66                 return Err(SystemError::InvalidRequest {
 67                     error: format!("Parsing query request: {}", e),
 68                     request: bin_request.into(),
 69                 })
 70             }
 71         };
 72         self.handle_query(&request)
 73     }
 74 }
 75 
 76 impl WasmMockQuerier {
 77     pub fn handle_query(&self, request: &QueryRequest<Empty>) -> QuerierResult {
 78         match &request {
 79             QueryRequest::Wasm(WasmQuery::Smart { contract_addr, msg }) => {
 80                 match from_binary(&msg).unwrap() {
 81                     Cw20QueryMsg::TokenInfo {} => {
 82                         let balances: &HashMap<HumanAddr, Uint128> =
 83                             match self.token_querier.balances.get(contract_addr) {
 84                                 Some(balances) => balances,
 85                                 None => {
 86                                     return Err(SystemError::InvalidRequest {
 87                                         error: format!(
 88                                             "No balance info exists for the contract {}",
 89                                             contract_addr
 90                                         ),
 91                                         request: msg.as_slice().into(),
 92                                     })
 93                                 }
 94                             };
 95 
 96                         let mut total_supply = Uint128::zero();
 97 
 98                         for balance in balances {
 99                             total_supply += *balance.1;
100                         }
101 
102                         Ok(to_binary(&TokenInfoResponse {
103                             name: "mAAPL".to_string(),
104                             symbol: "mAAPL".to_string(),
105                             decimals: 6,
106                             total_supply: total_supply,
107                         }))
108                     }
109                     Cw20QueryMsg::Balance { address } => {
110                         let balances: &HashMap<HumanAddr, Uint128> =
111                             match self.token_querier.balances.get(contract_addr) {
112                                 Some(balances) => balances,
113                                 None => {
114                                     return Err(SystemError::InvalidRequest {
115                                         error: format!(
116                                             "No balance info exists for the contract {}",
117                                             contract_addr
118                                         ),
119                                         request: msg.as_slice().into(),
120                                     })
121                                 }
122                             };
123 
124                         let balance = match balances.get(&address) {
125                             Some(v) => *v,
126                             None => {
127                                 return Ok(to_binary(&Cw20BalanceResponse {
128                                     balance: Uint128::zero(),
129                                 }));
130                             }
131                         };
132 
133                         Ok(to_binary(&Cw20BalanceResponse { balance }))
134                     }
135                     _ => panic!("DO NOT ENTER HERE"),
136                 }
137             }
138             _ => self.base.handle_query(request),
139         }
140     }
141 }
